#+TITLE: Shell scripting

* Basic template

#+BEGIN_SRC sh
#!/usr/bin/env bash
main(){
    readonly input_folder=input
    readonly output_folder=output
    create_folders
}

create_folders(){
    mkdir -p \
        "${input_folder}" \
        "${output_folder}"
}

main
#+END_SRC
